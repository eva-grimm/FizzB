                        // Gets the numbers from the value input fields
                        function getValues() {
                            // get the values from the form and attempt to convert string to number
                            let fizzNumber = Number(document.getElementById('fizz-value').value);
                            let buzzNumber = Number(document.getElementById('buzz-value').value);
                            let stopNumber = Number(document.getElementById('stop-value').value);

                            // validation
                            if (isNaN(fizzNumber) == true || isNaN(buzzNumber) == true || isNaN(stopNumber) == true) {
                                // if we don't have numbers
                                Swal.fire({
                                    title: 'Oops!',
                                    text: 'Sorry, but \'Fizz, B\' only works with real numbers',
                                    icon: 'error',
                                    backdrop: false
                                });
                            } else if (stopNumber < 1) {
                                // if we have a stop value less than 1
                                Swal.fire({
                                    title: 'Oops!',
                                    text: 'Sorry, but no stop value less than 1',
                                    icon: 'error',
                                    backdrop: false
                                });
                            } else if (stopNumber > 5000) {
                                // if we have a stop value greater than 5000
                                Swal.fire({
                                    title: 'Oops!',
                                    text: 'Sorry, but no stop value greater than 5000',
                                    icon: 'error',
                                    backdrop: false
                                });
                            } else {
                                // create an array of the numbers from 1 to stopNumber
                                let numbersToStyle = generateNumbers(stopNumber);

                                // convert numbersToStyle into HTML and push it to the applet page
                                displayNumbers(fizzNumber, buzzNumber, numbersToStyle);
                            }
                        }

                        function generateNumbers(stopNumber) {
                            let numbersToStyle = [];

                            // create an array of numbers from 1 up to stopNumber
                            for (let i = 1; i <= stopNumber; i++) {
                                numbersToStyle.push(i);
                            }

                            return numbersToStyle;
                        }

                        // Puts the numbers on the page
                        function displayNumbers(fizz, buzz, numbersToStyle) {
                            let tableHtml = '';
                            let currentHtmlString = '';

                            for (let i = 0; i < numbersToStyle.length; i++) {
                                // wipe currentHtmlString clean
                                currentHtmlString = '';
                                // set currentNumber to the next number in the array
                                let currentNumber = numbersToStyle[i];

                                // if the currentNumber is one greater than a number divisible by 5 
                                // or otherwise the number 1, then begin a new row
                                if (currentNumber % 5 == 1) {
                                    currentHtmlString = `<tr>`;
                                }

                                if (currentNumber % fizz == 0 && currentNumber % buzz == 0) {
                                    // if currentNumber is divisible by fizz *AND* buzz
                                    currentHtmlString += '<td class = fizz-buzz>Fizz, B!</td>'
                                } else if (currentNumber % fizz == 0) {
                                    // if currentNumber is divisible by *ONLY* fizz
                                    currentHtmlString += '<td class = fizz>Fizz</td>'
                                } else if (currentNumber % buzz == 0) {
                                    // if currentNumber is divisible by *ONLY* buzz
                                    currentHtmlString += '<td class = buzz>B</td>'
                                } else {
                                    // if currentNumber is divisible by *NEITHER* fizz nor buzz
                                    currentHtmlString += `<td>${currentNumber}</td>`
                                }

                                // if currentNumber is disivible by 5, then end the row
                                if (currentNumber % 5 == 0) {
                                    currentHtmlString += `</tr>`;
                                }

                                // append the string the loop's been building to tableHtml 
                                tableHtml += currentHtmlString;
                            }

                            document.getElementById('results').innerHTML = tableHtml;
                        }

'Fizz, B' is structured in three functions.

getValues

This function gets the values passed from the website form and attempts to convert them to numbers. From there, it validates that the values are valid for running the rest of the function. There are three checks in total:

  • Were numbers passed? Although the form is set to only allow numbers, some browsers may not respect that limitation, resulting in the user providing values that aren't numbers.
  • Is the stop value less than the 1? This applet requires a stop value of at least 1, so we need to confirm the stop value is 1 or more.
  • Is the stop value greater than 5,000? Because browsers can only handle so much calculation at once, I had to cap the stop value somewhere. The number 5,000 was selected arbitrarily but serves to give a wide range of choice for the user.

Once these checks have been performed, the function then calls generateNumbers(), passing it the stop value and receiving back an array. The function then passes that array on to displayNumbers().

generateNumbers

This function is the simplest but is no less integral. It receives the stop value as an argument, creates a blank array, then uses a for-loop to fill that array with numbers. The loop sets its index to 1 and runs iteratively until the index is greater than the ending number. While running, the loop pushes the current index into the array, which ultimately results in an array filled with numbers beginning with 1 and increasing by steps of 1 until the stop value is reached. If the chosen stop value is 1, the loop will run only once, resulting in an array with a single value of 1.

Once the array has been generated, the function returns it to getValues().

displayNumbers

The final function of the applet, this function's purpose is to render the array into html to be displayed on the applet page while specially formatting select numbers in the array. It receives the fizz value, buzz value, and the array generated earlier as arguments and begins by initializing two variables as blank strings.

  • The first variable is the table HTML that will be ultimately be passed to the applet page to display.
  • The second variable is the current string of HTML that will be built up over the course of the for-loop described below and which will be appended to the table HTML string at the conclusion of each instance of the loop.

Once the function has initialized its variables, it begins a for-loop. The loop begins with an index of 0 and works iteratively until the index is equal to the length of the passed array. The loop begins by resetting the current HTML string to blank, wiping the slate clean from the previous iteration. It then initializes a variable as the current number, the value is set to the value of the array at the current index of the array. It then begins constructing the html to be passed to the website in three steps:

  • Beginning a row: The function verifies whether the current number is one greater than a number divisible by 5 or otherwise the number 1. It does this by checking if the modulus of the current number is 1. If that is the case, then the function appends HTML code to the current string that begins a new row.
  • Checking the number: The function next checks whether the current number is divisible by the fizz value, buzz value, or both. If any of these circumstances is met, the function appends stylized HTML code to the string for the given case. If none of these circumstances is met, then the function instead appends basic HTML code to the current string that represents the current number in the table.
  • Ending a row: The function finally checks whether the current number is divisible by 5 by checking whether the modulus of the current number is 0. If that is the case, the function appends HTML code to the current string that ends the current row.

Once the current string has been fully constructed, the for-loop concludes by appending the current string to the end of the table string. Once the loop has concluded (and therefore the table string is fully constructed), the function uses DOM manipulation to change the 'results' element on the applet page to match the table string.
